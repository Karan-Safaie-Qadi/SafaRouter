Vanilla JS Integration

Using SafaRouter with plain JavaScript — no framework needed.

Minimal Vanilla JS Setup

SafaRouter shines in vanilla JS projects. Create a router, point it at a container, and load HTML pages. 

<!DOCTYPE html>
<html>
<head>
  <title>My SPA</title>
</head>
<body>
  <nav>
    <a href="/" data-safa-link>Home</a>
    <a href="/about" data-safa-link>About</a>
    <a href="/contact" data-safa-link>Contact</a>
  </nav>
  <main id="app"></main>

  <script type="module">
    import { SafaRouter } from 'safa-router'

    const router = new SafaRouter({
      target: '#app',
      pageDir: 'pages',
      routes: {
        '/': {},
        '/about': {},
        '/contact': {},
      },
    })

    router.start()
  </script>
</body>
</html>

How It Works

  1. Create an HTML element (e.g. <main id="app">) as the router target
  2. Define routes as a plain nested object — each key is a URL path
  3. The router loads matching HTML files from pageDir and injects them into the target
  4. Use data-safa-link on anchor tags for client-side SPA navigation
  5. Call router.start() to begin intercepting clicks and handling popstate

Smart Components in Vanilla JS

Smart components are plain functions that return HTML strings. They receive route context and re-render on every navigation. 

// header.js
export default function header({ path, params, query }) {
  const links = [
    { href: '/', label: 'Home' },
    { href: '/about', label: 'About' },
    { href: '/blog', label: 'Blog' },
  ]
  return [
    '<header style="border-bottom: 1px solid var(--color-border); padding: 0.75rem 1.5rem;">',
    '  <nav style="display: flex; gap: 1rem;">',
    ...links.map(({ href, label }) =>
      `    <a href="${href}" data-safa-link` +
      `      style="color: ${path === href ? 'var(--color-accent)' : 'var(--color-text-muted)'}">` +
      `      ${label}` +
      `    </a>`
    ),
    '  </nav>',
    '</header>',
  ].join('\n')
}

// main.js
import { SafaRouter } from 'safa-router'
import header from './header.js'

const router = new SafaRouter({
  target: '#app',
  pageDir: 'pages',
  components: { header },
  routes: {
    '/': {},
    '/about': {},
    '/blog': {
      children: { '[slug]': {} },
    },
  },
})

router.start()

Complete SPA Example

A full working SPA with nested layouts, dynamic routes, and middleware — all in vanilla JS: 

import { SafaRouter } from 'safa-router'
import header from './components/header.js'
import footer from './components/footer.js'

const router = new SafaRouter({
  target: '#app',
  basePath: '/my-app',
  pageDir: 'html-pages',
  components: { header, footer },

  // Nested layouts
  layout: 'html-pages/_layout.html',

  // Error handling
  errors: {
    pageDir: 'html-pages/errors',
    stackTraces: true,
  },

  // Access control
  access: {
    blocked: ['/admin', '/private/**'],
    ignored: ['/hidden'],
  },

  routes: {
    '/': {},
    '/about': {},
    '/blog': {
      children: { '[slug]': {} },
    },
    '/dashboard': {
      children: { settings: {} },
    },
  },
})

// Middleware
router.use(async (ctx, next) => {
  console.log(`Navigating to: ${ctx.path}`)
  return next()
})

// After-hook
router.afterEach(({ pathname }) => {
  document.title = `My App — ${pathname}`
})

// Error handler
router.onError(({ path, error, statusCode }) => {
  console.error(`[${statusCode}] ${path}:`, error.message)
})

router.start()

CDN Script (No Bundler)

Include SafaRouter directly from a CDN via ESM import map or script tag: 

<!-- index.html -->
<script type="importmap">
{
  "imports": {
    "safa-router": "https://unpkg.com/safa-router/dist/safa-router.js"
  }
}
</script>
<script type="module">
  import { SafaRouter } from 'safa-router'

  const router = new SafaRouter({
    target: '#app',
    pageDir: 'pages',
    routes: { '/': {}, '/about': {} },
  })

  router.start()
</script>
